# ===================================
# ИНСТРУКЦИИ ПО УСТАНОВКЕ
# ===================================

# 1. Клонировать репозиторий:
#    git clone <repository-url>
#    cd max_bot

# 2. Создать .env файл:
#    cp example-env .env
#    # Отредактируйте .env и укажите свои значения

# 3. Запустить через Docker Compose (рекомендуется):
#    docker-compose up -d

# 4. Или установить зависимости локально:
#    cd api && npm install
#    cd ../bot && npm install
#    cd ../web && npm install

# ===================================
# КОМАНДЫ ДЛЯ ЗАПУСКА
# ===================================

# Через Docker Compose (Production):
# docker-compose up -d

# Через Docker Compose (Development):
# docker-compose up

# Локальный запуск (Development):
# Терминал 1: cd api && npm run dev
# Терминал 2: cd bot && npm run dev
# Терминал 3: cd web && npm run dev

# Сборка Docker образов:
# docker-compose build

# Остановка сервисов:
# docker-compose down

# Просмотр логов:
# docker-compose logs -f
